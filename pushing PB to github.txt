    agent any

    stages {
        stage('Checkout') {
            steps {
                // Checkout the code from your Git repository
                git credentialsId: 'Github', url: 'https://github.com/Vaidhiyanathangithub/devopsproject.git'
            }
        }

        stage('Push to GitHub') {
            steps {
                script {
                    // Copy your playbook to the repository directory
                    sh 'cp /home/vaidhi/Documents/Playbooks/devopsproject.yml'

                    // Add, commit, and push the playbook
                    sh 'git add .'
                    sh 'git commit -m "Update playbook"'
                    sh 'git push origin master'
                }
            }
        }
    }
}
